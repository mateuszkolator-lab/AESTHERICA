## Kroki wgrania

### 1. Zaloguj się na serwer
```bash
ssh ubuntu@aestheticamd.ovh
cd /opt/aesthetica-md/deployment
```

### 2. Zatrzymaj kontenery
```bash
docker-compose down
```

### 3. Zaktualizuj pliki backendu

**server.py** - dodaj dotenv na początku:
```bash
cat > backend/server.py << 'EOF'
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
import os
from dotenv import load_dotenv

# Load environment variables from .env file
load_dotenv()

# Import routers
from routers.auth import router as auth_router
from routers.patients import router as patients_router
from routers.surgery_slots import router as surgery_slots_router
from routers.settings import router as settings_router
from routers.stats import router as stats_router
from routers.utils import router as utils_router
from routers.calendar import router as calendar_router

# Import database
from models.database import get_client

app = FastAPI(
    title="AestheticaMD API",
    description="Patient Management System for Facial Plastic Surgery",
    version="2.0.0"
)

# API Router with /api prefix
api_router = FastAPI()

# Include all routers
api_router.include_router(auth_router)
api_router.include_router(patients_router)
api_router.include_router(surgery_slots_router)
api_router.include_router(settings_router)
api_router.include_router(stats_router)
api_router.include_router(utils_router)
api_router.include_router(calendar_router)

# CORS
cors_origins = os.environ.get("CORS_ORIGINS", "*")
origins = cors_origins.split(",") if cors_origins != "*" else ["*"]

api_router.add_middleware(
    CORSMiddleware,
    allow_origins=origins,
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# Mount API router
app.mount("/api", api_router)

@app.get("/")
async def root():
    return {"message": "AestheticaMD API", "version": "2.0.0"}

@app.get("/health")
async def health():
    return {"status": "healthy"}
EOF
```

### 4. Przebuduj i uruchom
```bash
docker-compose build --no-cache
docker-compose up -d
```

### 5. Sprawdź logi
```bash
docker-compose logs -f backend
```
